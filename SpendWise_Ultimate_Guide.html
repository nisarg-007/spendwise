-- ── ENGINES ──────────────────────────────────────────────────
create table accounts (
    id uuid primary key default gen_random_uuid(),
    user_id uuid references auth.users(id) on delete cascade not null,
    type text check (type in ('bank', 'credit')),
    name text,
    balance numeric(12, 2) default 0,
    credit_limit numeric(12, 2),
    theme_idx int,
    color text
);

create table transactions (
    id uuid primary key default gen_random_uuid(),
    user_id uuid references auth.users(id) on delete cascade not null,
    account_id uuid references accounts(id),
    amount numeric(12, 2),
    type text check (type in ('income', 'expense')),
    category text,
    note text,
    date date default current_date,
    recurring boolean default false,
    tax_deductible boolean default false
);

create table budgets (
    id uuid primary key default gen_random_uuid(),
    user_id uuid references auth.users(id) not null,
    category text not null,
    amount numeric(12, 2) not null,
    month text not null
);

-- Row Level Security
alter table accounts enable row level security;
alter table transactions enable row level security;
create policy "user_owns_data" on accounts for all using (auth.uid() = user_id);
create policy "user_owns_data" on transactions for all using (auth.uid() = user_id);

Vercel Deployment Guide

To ensure a flawless deployment on Vercel and prevent the "Blank White Screen" or "Build Failed" errors, you MUST configure the following three settings in your Vercel Dashboard before your first deploy:

1. Root Directory

If your React app is inside a sub-folder (like spendwise/), go to Settings → General → Root Directory and enter the folder name. Otherwise, Vercel won't find your package.json.

2. Ignore Warnings

Vercel strictly fails builds if they have minor code warnings. Go to Settings → Environment Variables and add a new variable with Key: CI and Value: false.

3. Supabase Keys

Since your .env.local is hidden from GitHub, you must manually add your database keys in Settings → Environment Variables: REACT_APP_SUPABASE_URL and REACT_APP_SUPABASE_ANON_KEY.
